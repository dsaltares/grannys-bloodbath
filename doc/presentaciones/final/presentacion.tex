\section{Dificultades}

\begin{frame}
	\frametitle{Motor de colisiones}
	
	\begin{center}
		\includegraphics[scale=0.3]{images/colisiones.png}
	\end{center}
	
	\begin{center}
		De lo más complicado en un juego de plataformas
	\end{center}
	
	\begin{block}{Las pegas}
		\begin{itemize}
			\item Querer hacerlo demasiado preciso, es mejor el engaño
			\item Imposible controlar el uso que se haga del editor
		\end{itemize}
	\end{block}
	
	\begin{block}{A pesar de todo}
		\begin{itemize}
			\item Responde correctamente
			\item Es bastante flexible
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Manejo de escenas}
	
	\begin{center}
		\includegraphics[scale=0.07]{images/fondo-intro.png}
		\includegraphics[scale=0.07]{images/escena-barrio.png}
		\includegraphics[scale=0.07]{images/escena-boss.png}
	\end{center}
	
	\begin{block}{Los problemas}
		\begin{itemize}
			\item Querer que se ampliara con libertad y facilidad
			\item Totalmente externo al código fuente
			\item Manejo responsable de la memoria
		\end{itemize}
	\end{block}
	
	\begin{block}{La solución}
		\begin{itemize}
			\item SceneMngr (tratamiento uniforme para los tipos de escenas)
			\item Memoria dinámica (sólo los elementos necesarios)
			\item Ficheros XML
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Enemigos}
	
	\begin{center}
		\includegraphics[scale=0.17]{images/spider.png}
	\end{center}
	
	\begin{block}{Los problemas}
		\begin{itemize}
			\item Querer zombies tontos hasta cierto punto
			\item Buscar presentar un reto al jugador
		\end{itemize}
	\end{block}
	
	\begin{block}{La solución}
		\begin{itemize}
			\item Los zombies pueden acceder a su entorno (colisiones, otros actores)
			\item Creación de enemigos para presentar retos jugables
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Port a PSP}
	
	\begin{center}
		\includegraphics[scale=0.12]{images/portpsp.png}
	\end{center}
	
	\begin{block}{Los problemas}
		\begin{itemize}
			\item SDL soportada al 90\%
			\item Escasez de documentación
			\item Librerías no oficiales
			\item Memoria y procesador limitados
		\end{itemize}
	\end{block}
	
	\begin{block}{La solución}
		\begin{itemize}
			\item Creación de documentación :D
			\item Uso transparente del SDK en ocasiones
			\item Estamos trabajando en ello :(...
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Versión para Mac}
	
	\begin{center}

	\begin{block}{Los problemas}
