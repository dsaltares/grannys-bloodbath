\section{Introducción}

\begin{frame}
	\frametitle{¿De qué va Granny's Bloodbath?}
	
	\begin{block}{Historia}
	La protagonista es una abuelita octogenaria que vive plácidamente en su casita situada en una agradable zona residencial. 
	Todo iba de maravilla hasta que se ve invadida por terribles zombis
	\end{block}
	
	\pause
	
	\begin{block}{Mecánica}
		\begin{itemize}
			\item Plataformas de acción sorteando obstáculos y eliminando a todos los zombies que podamos
			\item Munición y powerups (dentadura y pastillas para la tensión)
			\item Ampliación de niveles por la comunidad
		\end{itemize}
	\end{block}
\end{frame}

\section{Problemas}

\begin{frame}
	\frametitle{Las colisiones, el mayor obstáculo}
	
	\begin{block}{Hemos tenido que tener en cuenta}
		\begin{itemize}
			\item Es un juego de plataformas, la detección de colisiones debe ser precisa
			\item Los personajes varían su superficie según su estado (andando, agachados...)
			\item Los elementos del escenario pueden ser atravesables, no atravesables o plataformas
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Solución}
	
	\begin{block}{Detección de colisiones}
		\begin{itemize}
			\item Los personajes tienen cajas que identifican sus zonas colisionables
			
			\begin{center}
				\includegraphics[scale=0.4]{colisiones-1.png}
			\end{center}
		\end{itemize}
	\end{block}
\end{frame}

	\frametitle{Solución}
